Benchmark¶

  • GitHub: https://github.com/google/benchmark

Installation¶

git clone --depth 1 https://github.com/google/benchmark.git
cd benchmark
mkdir build
cd build
cmake .. -DCMAKE_BUILD_TYPE=Release -DBENCHMARK_ENABLE_TESTING=OFF
make -j10

The generated libraries are in the directory build/src.

To build a program with benchmark, use:

  • compiler flag: -Iinclude -std=c++11

  • linker flag -Lbuild/src -lbenchmark

Warning

For gcc, we have to add -pthread to the linker flag. Note that it is NOT -lpthread.

#include <unistd.h>  // for usleep

#include "benchmark/benchmark.h"

namespace {

int test() {
  // usleep(10 * 1000);  // sleep 10ms
}

}  // namespace

static void BM_hello(benchmark::State& state) {
  for (auto _ : state) {
    test();
  }
}

BENCHMARK(BM_hello);

BENCHMARK_MAIN();

CXXFLAGS := -std=c++11
CXXFLAGS += -I$(HOME)/open-source/benchmark/include
LDFLAGS := -L$(HOME)/open-source/benchmark/build/src -lbenchmark
LDFLAGS += -pthread

srcs := $(shell find . -type f -name "*.cc" -printf %f)
targets :=$(srcs:%.cc=%)

all: $(targets)

%: %.cc
	$(CXX) $(CXXFLAGS) -o $@ $< $(LDFLAGS)
