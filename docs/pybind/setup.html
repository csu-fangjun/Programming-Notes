Build with setup.py¶

#include <pybind11/pybind11.h>

int add(int i, int j) { return i + j; }

PYBIND11_MODULE(hello, m) {
  m.doc() = "pybind11 hello world";  // optional module docstring

from setuptools import setup
from setuptools import Extension

import pybind11

# refer to https://docs.python.org/3/distutils/apiref.html#distutils.core.Extension
ext_modules = [
    Extension(name='hello',
              sources=['hello.cc'],
              extra_compile_args=['-std=c++11'],
              include_dirs=[pybind11.get_include()],
              language='c++'),
]

setup(name='hello-world-package',
      version='1.0',
      description='hello world in pybind11',
      author='fangjun',
      author_email='fangjun dot kuang at gmail dot com',
      url='https://github.com/csu-fangjun',
      long_description='This is really just a demo package.',
      ext_modules=ext_modules)

Execute python setup.py build and it will generate build/lib.linux-x86_64-3.5/hello.cpython-35m-x86_64-linux-gnu.so.

>>> import sys
>>> sys.path.insert(0, 'build/lib.linux-x86_64-3.5')
>>> import hello
>>> hello.add(1, 2)
3
