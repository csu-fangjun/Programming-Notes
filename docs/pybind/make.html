Build with Make¶

pybind11 is a header only library. To let the compiler know the places where pybind11 header files are, use

python3 -m pybind11 --includes

It should print something like:

-I/path/to/py35/include/python3.5m -I/path/to/py35/include/site/python3.5

In addition, the generated library has a special suffix for the filename. Something like .cpython-35m-x86_64-linux-gnu.so, which can be obtained by

python3-config --extension-suffix

To compile the following code,

hello.cc¶

#include <pybind11/pybind11.h>

int add(int i, int j) { return i + j; }

PYBIND11_MODULE(hello, m) {
  m.doc() = "pybind11 hello world";  // optional module docstring

  m.def("add", &add, "A function which adds two numbers");
}

srcs := hello.cc

suffix := $(shell python3-config --extension-suffix)

CXXFLAGS := -std=c++11
CXXFLAGS += -shared -fPIC
CXXFLAGS += $(shell python3 -m pybind11 --includes)

name := hello$(suffix)

all: $(name)

$(name): $(srcs)
	$(CXX) $(CXXFLAGS) -o $@ $<

.PHONE: clean test

test: $(name)
	python3 -c "import hello; print(hello.add(1, 2))"

clean:
	-$(RM) $(name)

make all will generate a file whose name looks like hello.cpython-35m-x86_64-linux-gnu.so.

To use it in Python, modify PYTHONPATH and use import hello to import this module.
